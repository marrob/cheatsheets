### back-button
https://ionicframework.com/docs/api/back-button
Csak akkor jelenit meg a back buttont ha tudja hogy hová lehet visszalépni.
<!-- Default back button -->
<ion-header>
  <ion-toolbar>
    <ion-buttons slot="start">
      <ion-back-button></ion-back-button>
    </ion-buttons>
  </ion-toolbar>
</ion-header>

<!-- defaultHref -->
Ha nincs el?zmény akkor a defaultHref="/"-ben megadott linkre lép
<ion-header>
  <ion-toolbar>
    <ion-buttons slot="start">
      <ion-back-button defaultHref="/places/tabs/discover"></ion-back-button>
    </ion-buttons>
    <ion-title>Place Detail</ion-title>
  </ion-toolbar

constructor(public alertCtrl: AlertController) { }
              
async onDeleteRecpie() {
  const alert = await this.alertCtrl.create({
    header: 'Are you sure',
    subHeader: 'Do you really want to delete the recpie?',
    message: 'Your battery about to die',
    buttons: [
      {
        text:'Cancel',
        role:'cancel'
      },
      {
        text:'Delete',
        handler:() => { 
          /*Do Something*/
        }
      }
    ]
  });
  await alert.present();
  const result = await alert.onDidDismiss();
  console.log(result);
}
